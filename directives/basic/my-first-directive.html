My First Directive

In this example we are creating a simple directive which uses only 1 property called 'template'

Code:

                

                    /////////////
                    //Angular App
                    ////////////
                    var myApp = angular.module("myApp", []);


                    /////////////
                    //Angular Controller
                    ////////////
                    myApp.controller('Controller1',[Controller1]);

                    function Controller1(){

                    }

                    /////////////
                    //Angular Directive
                    ////////////
                    myApp.directive('myFirstDirective',[MyFirstDirective]);

                    function MyFirstDirective(){
                        return{
                              template:'
Hello From My First Directive
'
                        }
                    }
