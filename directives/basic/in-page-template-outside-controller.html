Inpage Template using ng-template talking to Controllers

In this example we are creating a directive using ng-template reading variables of a outside controller

HTML:

                

                    <body>
                    .... ...

                        <div class="container" ng-controller="Controller2 as Ctrl">

                            <script type="text/ng-template" id="simple-directive-template.html">

                                <h5>Check this out:</h5>
                                <p>Message from Controller: {{  Ctrl.message  }} </p>

                            </script>

                            <h4>Result:</h4>
                            <my-inpage-template-directive></my-inpage-template-directive>

                        </div>

                    ... ...

                    </body>

JavaScript:

                

                    /////////////
                    //Angular App
                    ////////////
                    var myApp = angular.module("myApp", []);


                    /////////////
                    //Angular Controller
                    ////////////
                    myApp.controller('Controller2',[Controller2]);

                    function Controller2(){
                        this.message = "Hello from Controller 2";
                    }

                    /////////////
                    //Angular Directive
                    ////////////
                    myApp.directive('myInpageTemplateDirective',[MyInpageTemplateDirective]);

                    function MyInpageTemplateDirective(){
                        return {
                             templateUrl:'simple-directive-template.html'
                        }
                    }
